#include <math.h>
bool canMakeArithmeticProgression(int* arr, int arrSize) {
    for (int i = 0; i < arrSize; i++) {
        for (int j = i + 1; j < arrSize; j++) {
            if (arr[i] > arr[j])
                arr[i] ^= arr[j] ^= arr[i] ^= arr[j];
        }
    }

    int diff = abs(arr[1] - arr[0]);
    for (int k = arrSize-1; k > 1; k--) {
        if (abs(arr[k] - arr[k - 1]) != diff) {
            return false;
        }
    }

    return true;
}
