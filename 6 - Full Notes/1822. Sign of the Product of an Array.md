int signFunc(int x) {
    if (x < 0) {
        return -1;
    }
    if (x > 0) {
        return 1;
    } else {
        return 0;
    }
}
int arraySign(int* nums, int numsSize) {
    int prod = 1;
    for (int i = 0; i < numsSize; i++) {
        if (nums[i] == 0) {
            return (signFunc(0));
        } else if (nums[i] < 0) {
            nums[i] = -1;
        } else {
            nums[i] = 1;
        }

        prod *= nums[i];
    }
    return (signFunc(prod));
}
