double average(int* salary, int salarySize) {
    for (int i = 0; i < salarySize; i++) {
        for (int j = i; j < salarySize; j++) {
            if (salary[i] > salary[j]) {
                int temp = 0;
                temp = salary[i];
                salary[i] = salary[j];
                salary[j] = temp;
            }
        }
    }
    double sum = 0;
    for (int i = 1; i < salarySize - 1; i++) {
        sum += salary[i];
    }

    return sum/(salarySize-2);
}
